  <script>
    const startBtn = document.getElementById("startBtn");
    const toggleHistoryBtn = document.getElementById("toggleHistoryBtn");
    const clearBtn = document.getElementById("clearBtn");
    const wakeToggle = document.getElementById("wakeToggle");
    const soundToggle = document.getElementById("soundToggle");
    const gauge = document.getElementById("gauge");
    const angleValue = document.getElementById("angleValue");
    const angleLabel = document.getElementById("angleLabel");
    const statusText = document.getElementById("statusText");
    const warningText = document.getElementById("warningText");
    const historyPanel = document.getElementById("historyPanel");
    const storageKey = "tek-teker-egim-logs-v1";
    let permissionGranted = false;
    let wakeLockSentinel = null;
    let latestAngle = 0;
    let lastAlertAt = 0;
    let lastLogAt = 0;
    let logs = [];
    function loadLogs() { try { const raw = localStorage.getItem(storageKey); logs = raw ? JSON.parse(raw) : []; } catch { logs = []; } renderLogs(); }
    function saveLogs() { localStorage.setItem(storageKey, JSON.stringify(logs.slice(-20))); renderLogs(); }
    function addLog(angle) { const now = new Date(); logs.push({ time: now.toLocaleTimeString("tr-TR", { hour: "2-digit", minute: "2-digit", second: "2-digit" }), angle: Math.round(angle), state: classifyAngle(angle) }); if (logs.length > 20) logs.shift(); saveLogs(); }
    function classifyAngle(angle) { if (angle < 10) return "Çok sakin"; if (angle < 20) return "Düşük"; if (angle < 30) return "Orta"; if (angle < 45) return "Yüksek"; return "Tehlikeli"; }
    function renderLogs() { if (!logs.length) { historyPanel.innerHTML = '<div class="muted">Henüz ölçüm yok.</div>'; return; } historyPanel.innerHTML = logs.slice().reverse().map(function(item) { return '<div class="history-item"><span>' + item.time + '</span><span>' + item.angle + '° · ' + item.state + '</span></div>'; }).join(""); }
    function interpolateColor(start, end, t) { const mix = function(a, b, p) { return Math.round(a + (b - a) * p); }; return start.map(function(value, index) { return mix(value, end[index], t); }); }
    function colorForAngle(angle) { const clamped = Math.max(0, Math.min(angle, 90)); const ratio = clamped / 90; const white = [255, 255, 255]; const green = [150, 230, 160]; const yellow = [245, 215, 90]; const red = [255, 90, 90]; if (ratio < 0.35) return interpolateColor(white, green, ratio / 0.35); if (ratio < 0.70) return interpolateColor(green, yellow, (ratio - 0.35) / 0.35); return interpolateColor(yellow, red, (ratio - 0.70) / 0.30); }
    function updateDisplay(angle) { const safeAngle = Math.max(0, Math.min(90, Math.round(angle))); const rgba = colorForAngle(safeAngle); gauge.style.background = "linear-gradient(135deg, rgba(" + rgba[0] + ", " + rgba[1] + ", " + rgba[2] + ", 0.95), rgba(255,255,255,0.8))"; gauge.style.color = safeAngle > 45 ? "#fff" : "#0f172a"; angleValue.textContent = safeAngle + "°"; angleLabel.textContent = safeAngle >= 30 ? "Yüksek eğim" : "İzleniyor"; statusText.textContent = classifyAngle(safeAngle); warningText.textContent = safeAngle >= 30 ? "Uyarı" : "Normal"; if (safeAngle >= 30 && soundToggle.checked && Date.now() - lastAlertAt > 1200) { beep(); if ("vibrate" in navigator) navigator.vibrate([120, 50, 120]); lastAlertAt = Date.now(); } if (Date.now() - lastLogAt > 1000) { addLog(safeAngle); lastLogAt = Date.now(); } }
    function beep() { try { const Ctx = window.AudioContext || window.webkitAudioContext; if (!Ctx) return; const ctx = new Ctx(); const oscillator = ctx.createOscillator(); const gain = ctx.createGain(); oscillator.type = "square"; oscillator.frequency.setValueAtTime(880, ctx.currentTime); gain.gain.setValueAtTime(0.08, ctx.currentTime); gain.gain.exponentialRampToValueAtTime(0.0001, ctx.currentTime + 0.25); oscillator.connect(gain); gain.connect(ctx.destination); oscillator.start(); oscillator.stop(ctx.currentTime + 0.25); ctx.close(); } catch (err) {} }
    async function requestWakeLock() { if (!("wakeLock" in navigator) || !wakeToggle.checked) return; try { wakeLockSentinel = await navigator.wakeLock.request("screen"); } catch (err) {} }
    async function releaseWakeLock() { if (wakeLockSentinel) { try { await wakeLockSentinel.release(); } catch (err) {} wakeLockSentinel = null; } }
    async function startSensors() { if (permissionGranted) return; if (typeof DeviceOrientationEvent !== "undefined" && typeof DeviceOrientationEvent.requestPermission === "function") { try { const result = await DeviceOrientationEvent.requestPermission(); if (result !== "granted") { angleLabel.textContent = "İzin reddedildi"; return; } } catch (err) { angleLabel.textContent = "İzin alınamadı"; return; } } if (typeof DeviceMotionEvent !== "undefined" && typeof DeviceMotionEvent.requestPermission === "function") { try { const result = await DeviceMotionEvent.requestPermission(); if (result !== "granted") { angleLabel.textContent = "İzin reddedildi"; return; } } catch (err) { angleLabel.textContent = "İzin alınamadı"; return; } } permissionGranted = true; startBtn.textContent = "Ölçüm çalışıyor"; startBtn.disabled = true; angleLabel.textContent = "Sensörler aktif"; window.addEventListener("deviceorientation", handleOrientation); window.addEventListener("devicemotion", handleMotion); await requestWakeLock(); updateDisplay(latestAngle); }
    function handleOrientation(event) { if (event.beta !== null && Number.isFinite(event.beta)) latestAngle = Math.abs(event.beta); }
    function handleMotion(event) { if (!event.accelerationIncludingGravity) return; const ax = event.accelerationIncludingGravity.x || 0; const ay = event.accelerationIncludingGravity.y || 0; const az = event.accelerationIncludingGravity.z || 0; const angle = Math.abs(Math.atan2(ay, Math.sqrt(ax * ax + az * az)) * 180 / Math.PI); if (Number.isFinite(angle)) latestAngle = Math.max(0, Math.min(90, angle)); }
    startBtn.addEventListener("click", function() { startSensors(); });
    toggleHistoryBtn.addEventListener("click", function() { historyPanel.classList.toggle("hidden"); toggleHistoryBtn.textContent = historyPanel.classList.contains("hidden") ? "Geçmişi göster" : "Geçmişi gizle"; });
    clearBtn.addEventListener("click", function() { logs = []; localStorage.removeItem(storageKey); renderLogs(); });
    wakeToggle.addEventListener("change", function() { if (wakeToggle.checked) requestWakeLock(); else releaseWakeLock(); });
    document.addEventListener("visibilitychange", function() { if (document.visibilityState === "visible") requestWakeLock(); else releaseWakeLock(); });
    setInterval(function() { if (permissionGranted) updateDisplay(latestAngle); }, 250);
    if ("serviceWorker" in navigator) navigator.serviceWorker.register("./sw.js").catch(function() {});
    loadLogs(); renderLogs();
  </script>
